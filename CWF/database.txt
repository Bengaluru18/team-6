create table school_details( school_id integer(100) auto_increment primary key , school_name varchar(200) not null, locality varchar(20), address varchar(500), district varchar(200), state varchar(200), LPS int(3),HPS int(3),gender varchar(20), total_no_of_students int(4), MOI varchar(25), DISE varchar(20));
create schema cfw;
use cfw;
create table kannada_medium(school_id integer(100) ,class int(4) primary key, boys varchar(100), girls varchar(100), total int(4),foreign key(school_id) references school_details(school_id));
create table english_medium(school_id integer(100),class int(4) primary key, boys varchar(100), girls varchar(100), total int(4),foreign key(school_id) references school_details(school_id));
create table SC_ST_children(school_id integer(100), b_sc int(3),b_st int(3),g_sc int(3),g_st int(3),foreign key(school_id) references school_details(school_id));
create table teacher_count(school_id integer(100),male varchar(20),female varchar(20),fulltime varchar(20), total int(4),foreign key(school_id) references school_details(school_id));
create table teacher_details(school_id integer(100),designation varchar(200), name varchar(200), contact_number int(12), email varchar(200),foreign key(school_id) references school_details(school_id));
create table stakeholder(stakeholder_name varchar(200), total_members int(8), meeting_duration varchar(200), contact_details int(10));
create table individual_donor(donor_name varchar(200),total_members int(7), major_activities varchar(200), contact_details int(12));
show tables;
create table sanitation( school_id integer(100),toilet_details varchar(200), availability varchar(100), req_repair varchar(100),new_requirement varchar(100), aoir varchar(1000),foreign key(school_id) references school_details(school_id));
create table classroom_environment(  school_id integer(100), indicators varchar(200), total_available int(8),req_repair varchar(100),new_requirement varchar(100), aoir varchar(1000),foreign key(school_id) references school_details(school_id));
create table school_environment(school_id integer(100), room_name varchar(200), availablity varchar(100), total_no int(5), reno_require varchar(100), flooring varchar(100),plastering varchar(100), waterproofing varchar(100), electrification varchar(100), plumbing varchar(100), blackboard varchar(100), wall_mount varchar(100),safety_grill varchar(100), external_painting varchar(100), remarks varchar(500), foreign key(school_id) references school_details(school_id));
create table login_details(username varchar(100), pwd varchar(100), email varchar(200));
create table admin_details(username varchar(100), pwd varchar(100), email varchar(200));
create table ngo_details(username varchar(100), pwd varchar(100), email varchar(200));
